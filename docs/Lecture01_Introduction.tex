\subsection{What is International
Trade?}\label{what-is-international-trade}

\textbf{Definition}

\begin{quote}
International trade is the \textbf{cross-border exchange} of goods,
services, and capital between residents of different countries.
\end{quote}

\textbf{Three dimensions of trade}

\begin{itemize}
\tightlist
\item
  \textbf{Merchandise trade} --- physical goods: rice, wheat, machinery,
  petroleum
\item
  \textbf{Services trade} --- IT services, tourism, financial services,
  education
\item
  \textbf{Capital flows} --- foreign direct investment, portfolio
  investment, remittances
\end{itemize}

\textbf{Why it matters for agricultural economists}

\begin{itemize}
\tightlist
\item
  India earns \textasciitilde₹3.6 lakh crore from agricultural exports
  annually
\item
  Global price signals directly affect domestic farm prices
\item
  WTO rules constrain domestic agricultural support policies
\end{itemize}

\textbf{Core insight:} Trade is fundamentally about countries doing what
they are relatively better at, and exchanging the surplus --- creating
gains for all participants.

\subsection{Why Do Nations Trade?}\label{why-do-nations-trade}

\textbf{Supply-side reasons}

\begin{itemize}
\tightlist
\item
  \textbf{Resource endowments} --- India has fertile Indo-Gangetic
  plains; Brazil has Amazon basin → different agricultural outputs
\item
  \textbf{Technology differences} --- Netherlands uses precision
  horticulture; India relies on labour-intensive methods
\item
  \textbf{Specialization gains} --- Focusing on fewer products reduces
  unit cost
\item
  \textbf{Economies of scale} --- Large production runs lower average
  cost (e.g., Thai rice mills)
\end{itemize}

\textbf{Demand-side reasons}

\begin{itemize}
\tightlist
\item
  \textbf{Product variety} --- Consumers want Darjeeling tea \emph{and}
  Colombian coffee
\item
  \textbf{Seasonal complementarity} --- Southern hemisphere harvests
  when Northern hemisphere is off-season
\item
  \textbf{Income growth} --- Rising incomes raise demand for diverse
  food products
\end{itemize}

\textbf{Fundamental driver}

\begin{itemize}
\tightlist
\item
  Countries differ in \textbf{opportunity costs} of production
\item
  Differences in opportunity costs → gains from specialisation and trade
\end{itemize}

\section{Part III: Key Concepts and
Context}\label{part-iii-key-concepts-and-context}

\subsection{Key Concepts We Will Need}\label{key-concepts-we-will-need}

\textbf{Theory concepts}

\begin{itemize}
\tightlist
\item
  \textbf{Comparative advantage} --- trade based on \emph{relative}, not
  absolute, efficiency (Ricardo)
\item
  \textbf{Terms of trade} --- ratio of export prices to import prices;
  determines how much a country gains from trade
\item
  \textbf{Factor endowments} --- land, labour, capital abundance shapes
  what countries export (Heckscher-Ohlin)
\item
  \textbf{Intra-industry trade} --- two-way trade in similar products
  (India exports basmati rice, imports parboiled rice)
\end{itemize}

\textbf{Policy concepts}

\begin{itemize}
\tightlist
\item
  \textbf{Tariff} --- tax on imports; raises domestic price above world
  price
\item
  \textbf{Non-Tariff Barrier (NTB)} --- quotas, SPS standards, licensing
\item
  \textbf{Balance of Payments (BoP)} --- accounting of all international
  transactions
\item
  \textbf{Exchange rate} --- price of one currency in terms of another;
  a stronger rupee makes exports more expensive
\item
  \textbf{Trade balance} --- exports minus imports; positive = surplus,
  negative = deficit
\end{itemize}

\textbf{How these connect:} Comparative advantage explains \emph{what}
countries trade; terms of trade determine \emph{how much} they gain;
tariffs and NTBs determine \emph{how much} trade actually occurs.

\subsection{Why Study International
Trade?}\label{why-study-international-trade}

\textbf{Policy relevance for India}

\begin{itemize}
\tightlist
\item
  \textbf{Foreign Trade Policy 2023--28} targets \$2 trillion in
  merchandise + services exports by 2030

  \begin{itemize}
  \tightlist
  \item
    From \$777B (FY2023) to \$2T in 7 years --- requires
    \textasciitilde14.5\% annual growth
  \end{itemize}
\item
  \textbf{WTO commitments} constrain India's ability to protect domestic
  agriculture

  \begin{itemize}
  \tightlist
  \item
    Aggregate Measurement of Support (AMS) limits; tariff bindings
  \end{itemize}
\item
  \textbf{Farmer income} --- \textasciitilde50\% of India's workforce in
  agriculture; export markets expand income opportunities
\item
  \textbf{Food security} --- Import policy on pulses and oilseeds
  directly affects 1.4 billion people
\end{itemize}

\textbf{Academic relevance}

\begin{itemize}
\tightlist
\item
  Fastest-growing sub-field in development economics
\item
  Links to climate change (Scope 3 emissions in supply chains), gender
  (women in agri-export supply chains), and technology (digital trade in
  services)
\end{itemize}

\textbf{Bottom line:} International trade policy is \emph{agricultural
policy} in India. They cannot be studied in isolation.

\subsection{What Makes Agricultural Trade
Special?}\label{what-makes-agricultural-trade-special}

\textbf{Biological \& physical features}

\begin{itemize}
\tightlist
\item
  \textbf{Perishability} --- fresh produce, milk, meat have short
  shelf-lives → need cold chains, quick clearance
\item
  \textbf{Seasonality} --- harvests are discrete events; supply is lumpy
\item
  \textbf{Bulkiness} --- grain, sugar, cotton: high weight-to-value
  ratio → high freight costs
\item
  \textbf{Sanitary risks} --- food-borne pathogens cross borders with
  commodities → SPS measures justified
\end{itemize}

\textbf{Economic \& political features}

\begin{itemize}
\tightlist
\item
  \textbf{Food security concern} --- governments intervene to ensure
  domestic supply; export bans, import duty waivers are common
\item
  \textbf{Price volatility} --- weather, pests, and speculation create
  large price swings; farmers and consumers are both affected
\item
  \textbf{Political economy} --- large farm lobbies in US, EU, India
  influence trade policy disproportionately
\item
  \textbf{WTO special treatment} --- Agreement on Agriculture has
  special provisions for developing countries (Special and Differential
  Treatment)
\end{itemize}

\section{Part IV: Policy and
Frameworks}\label{part-iv-policy-and-frameworks}

\subsection{Recent Trade Policy
Highlights}\label{recent-trade-policy-highlights}

\textbf{Foreign Trade Policy (FTP) 2023--28: Key pillars}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  \textbf{Export target:} \$2 trillion merchandise + services exports by
  2030
\item
  \textbf{Districts as Export Hubs (DEH):} Identify 1 product per
  district with export potential; provide targeted support
\item
  \textbf{ECGC coverage:} Expanded Export Credit Guarantee Corporation
  cover to de-risk exporters, especially MSMEs
\item
  \textbf{Remission of Duties and Taxes on Exported Products (RoDTEP):}
  Replace old MEIS; neutralise embedded taxes
\item
  \textbf{E-commerce exports:} Dedicated policy for cross-border
  e-commerce; target \$200--300B by 2030
\end{enumerate}

\textbf{Agricultural trade--specific provisions}

\begin{itemize}
\tightlist
\item
  Agri-specific chapters supporting APEDA-registered exporters
\item
  GI tag promotion (Darjeeling tea, Alphonso mangoes, Basmati rice)
\item
  Simplification of phytosanitary certification processes
\end{itemize}

FTP 2023--28 is notably continuity-oriented --- it builds on previous
policy architecture rather than a radical overhaul, reflecting India's
cautious approach to trade liberalisation.

\subsection{A Preview: The Basic Trade
Model}\label{a-preview-the-basic-trade-model}

\textbf{Production Possibility Frontier (PPF) --- conceptual preview}

\textbf{Without trade (autarky)}

\begin{itemize}
\tightlist
\item
  A country produces \emph{and} consumes on its PPF
\item
  The slope of the PPF at the consumption point = the domestic relative
  price
\item
  Example: India producing only rice and wheat --- the trade-off between
  them is fixed by domestic technology
\end{itemize}

\textbf{With trade (open economy)}

\begin{itemize}
\tightlist
\item
  World price ratio differs from domestic price ratio
\item
  Country specialises in the good where it has comparative advantage
\item
  Consumption point can now lie \textbf{outside} the PPF
\item
  \emph{Both} countries gain --- consumption of both goods increases
\end{itemize}

\textbf{The core result:}

Trade allows a country to \textbf{consume beyond its production
possibilities} --- the equivalent of getting something for nothing,
achieved through specialisation and exchange.

\tightlist
\item

\item
  How much each country gains --- terms of trade (Lec 7)
\item
  What tariffs do to this picture (Lec 8)
\end{itemize}

\subsection{Key Definitions}\label{key-definitions}

\textbf{Trade flows}

\begin{itemize}
\tightlist
\item
  \textbf{Exports} --- goods/services sold to foreign residents; flow of
  goods out, money in
\item
  \textbf{Imports} --- goods/services bought from foreign residents;
  flow of goods in, money out
\item
  \textbf{Trade balance} = Exports − Imports

  \begin{itemize}
  \tightlist
  \item
    Positive → trade surplus
  \item
    Negative → trade deficit
  \end{itemize}
\item
  \textbf{Visible trade} --- physical merchandise (rice, cars, oil)
\item
  \textbf{Invisible trade} --- services, remittances, investment income
\end{itemize}

\textbf{Balance of Payments}

\begin{itemize}
\tightlist
\item
  \textbf{Current account} --- trade in goods + services + remittances +
  investment income
\item
  \textbf{Capital account} --- financial flows: FDI, portfolio
  investment, loans
\item
  \textbf{BoP identity:} Current account + Capital account = 0
  (approximately)
\end{itemize}

\textbf{Exchange rate}

\begin{itemize}
\tightlist
\item
  \textbf{Nominal exchange rate} --- ₹/USD; e.g., ₹83 = \$1
\item
  \textbf{Real exchange rate} --- adjusts for price level differences;
  determines \emph{actual} competitiveness
\item
  \textbf{Appreciation} → exports become more expensive → reduces export
  competitiveness
\end{itemize}

\subsection{Summary: Why International Trade
Matters}\label{summary-why-international-trade-matters}

\textbf{Four reasons international trade is essential for India}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\item
  \textbf{Growth engine:} Trade has been central to every development
  success story --- Japan, South Korea, China, and now India's IT sector
\item
  \textbf{Consumer welfare:} Trade gives 1.4 billion Indians access to
  goods they cannot produce cheaply at home --- edible oils,
  electronics, fertilisers
\item
  \textbf{Farmer incomes:} Export markets for rice, spices, and marine
  products support millions of rural households; global price
  transmission improves farm income signals
\item
  \textbf{Policy discipline:} WTO commitments and trade agreements
  create external pressure for domestic policy reform --- reducing
  market distortions that hurt productivity
\end{enumerate}

\textbf{India's challenge:} Integrating deeply into global value chains
while protecting food security and farm incomes --- the central tension
this course will explore.
